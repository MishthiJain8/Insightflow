\begin{frame}{What is InsightFlow?}
\begin{block}{Mission Statement}
A \textbf{self-hosted, AI-powered financial intelligence platform} that helps retail investors make data-driven decisions using machine learning, NLP, and multimodal analysis --- with \textbf{zero paid API costs}.
\end{block}

\vspace{0.3cm}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{What It Does:}
\begin{itemize}
    \item Analyzes stocks, crypto, and indices
    \item Predicts price direction using ML
    \item Validates predictions and self-corrects
    \item Explains outputs in plain English
    \item Tracks portfolios with AI alerts
\end{itemize}

\column{0.48\textwidth}
\textbf{Who It Helps:}
\begin{itemize}
    \item Retail investors new to trading
    \item Quantitative finance enthusiasts
    \item Anyone wanting AI stock insights
    \item Students learning ML in finance
    \item Privacy-focused self-hosted users
\end{itemize}
\end{columns}

\begin{frame}{How It Works --- 10-Stage Query Pipeline}

\begin{center}
\small
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{c l l}
\toprule
\textbf{Stage} & \textbf{Component} & \textbf{What Happens} \\
\midrule
1 & User Input & User types: ``\textit{Should I invest in Apple?}'' \\
2 & NLP Parser & Extracts ticker (AAPL), intent (buy), horizon (7d) \\
3 & Zero-Shot Classifier & BART-MNLI confirms intent category \\
4 & Market Data & yfinance pulls OHLCV, Fundamentals, Options \\
5 & Feature Engine & Computes 16 Features (TA, ^VIX, P/E, Put/Call) \\
6 & XGBoost Ensemble & Predicts direction (UP/DOWN) via voted probability \\
7 & FinBERT Sentiment & Analyzes last 8 news headlines for sentiment bias \\
8 & wav2vec2 Audio & Analyzes audio context for emotional tone \\
9 & Signal Fusion & Combines text + audio + technicals into composite \\
10 & Llama3 RAG Engine& Generates conversational RAG explanation + action \\
\bottomrule
\end{tabular}
\end{center}

\begin{block}{Example Output}
Direction: \textbf{UP} (68\% confidence) \quad Sentiment: \textbf{Bullish} \quad Action: \textbf{BUY} \quad Target: \textbf{\$198.50}
\end{block}

\begin{frame}{How It Works --- Self-Correction Feedback Loop}

\begin{center}
\small
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{c l l}
\toprule
\textbf{Step} & \textbf{Phase} & \textbf{What Happens} \\
\midrule
1 & Predict & Model makes a directional prediction (UP/DOWN) \\
2 & Store & Prediction logged to MongoDB with target date \\
3 & Wait & System waits for the horizon period (e.g., 7 days) \\
4 & Evaluate & Compares predicted vs actual price movement \\
5 & Score & Updates rolling 20-prediction accuracy tracker \\
6 & Adjust & Modifies model hyperparameters and feature weights \\
\bottomrule
\end{tabular}
\end{center}

\vspace{0.3cm}
\begin{columns}[T]
\column{0.48\textwidth}
\begin{alertblock}{If Accuracy $<$ 60\%}
\begin{itemize}
    \item Escalate XGBoost/RF Model complexity
    \item n\_estimators: 200 $\rightarrow$ 300 $\rightarrow$ 400
    \item max\_depth: 6 $\rightarrow$ 8 $\rightarrow$ 10
    \item Penalize misleading features
\end{itemize}
\end{alertblock}

\column{0.48\textwidth}
\begin{exampleblock}{If Accuracy $\geq$ 70\%}
\begin{itemize}
    \item Relax parameters toward defaults
    \item Reduce overfitting risk
    \item Reward accurate feature weights
    \item Generate AI learning notes
\end{itemize}
\end{exampleblock}
\end{columns}
